# Dépendances pour le système de cache Redis

# Django Redis - Backend de cache Redis pour Django
# Installation: pip install django-redis
django-redis>=5.4.0

# Redis Python Client (installé automatiquement avec django-redis)
redis>=5.0.0

# Note: Pour utiliser le cache en développement sans Redis,
# aucune dépendance supplémentaire n'est nécessaire.
# Django utilise automatiquement LocMemCache si REDIS_URL n'est pas défini.

# Installation Redis (serveur):
# Windows: https://github.com/microsoftarchive/redis/releases
# Linux: sudo apt-get install redis-server
# Mac: brew install redis

# Démarrer Redis:
# Windows: redis-server.exe
# Linux/Mac: redis-server

# Tester Redis:
# redis-cli ping
# (devrait retourner PONG)
